subroutine LOAD

st=>start: Start e=>end get_para=>operation: get parameters by Head type: z pos1 z pos2 mov_z1=>subroutine: move Z : pos = pos1 speed = speed1 mov_z2=>subroutine: move Z : pos = pos2 speed = speed2 st->get_para->mov_z1->mov_z2->e

subroutine UNLOAD

st=>start: Start e=>end get_para=>operation: get parameters by Head type: z pos mov_z1=>subroutine: move Z : pos = pos1 speed = speed1 mov_z2=>subroutine: move P : pos = unload pos ( - xxx) home=>subroutine: move P: pos = 0 st->get_para->mov_z1->mov_z2->home->e

subroutine AIR GAP

st=>start: Start e=>end get_para=>operation: get parameters by Head type,Plate type,settings: p pos for air gap mov_p_air=>subroutine: move P: pos = pos for air gap st->get_para->mov_p_air->e

subroutine PICKUP

st=>start: Start e=>end get_para=>operation: get parameters by Head type,Plate type,settings: z pos1 p pos2 mov_z1=>subroutine: move Z : pos = pos1 mov_p=>subroutine: move P : pos = pos2 for liquid st->get_para->mov_z1->mov_p->e

subroutine DISPENSE

st=>start: Start e=>end get_para=>operation: get parameters by Head type,Plate type,settings: z pos1 p pos2 (volume) mov_z1=>subroutine: move Z : pos = pos1 mov_p=>subroutine: move P : pos = pos - pos2 for liquid st->get_para->mov_z1->mov_p->e

subroutine MOVE Z

st=>start: Start e=>end check_lh=>condition: LEFT HALL == 1? set_rang_1=>operation: range = bottom check_rh=>condition: RIGHT HALL == 1? set_rang_2=>operation: range = up check_range=>condition: check range check_pos=>condition: @position? error=>operation: Error check_motor=>condition: motor error? step=>operation: Motor Step st->check_lh(yes)->set_rang_1->check_range(yes)->step->check_pos(yes)->e check_range(no)->error check_lh(no)->check_rh(yes)->set_rang_2->check_range check_rh(no)->error check_pos(no)->check_motor(no,left)->step check_motor(yes)->error error->e
